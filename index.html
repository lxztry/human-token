💻 费率配置

const RATES = {
  "废话文学": 0.001,      // 如"啊这个那个嗯"
  "普通对话": 0.01,       // 日常聊天
  "深度思考": 0.1,        // 讨论问题/解决问题
  "情绪价值": 0.5,        // 安慰、鼓励、倾听
  "老板发言": 0,          // 免费！（潜规则）
}

🚀 快速开始

# 克隆项目
git clone https://github.com/lxztry/human-token.git
cd human-token

# 安装依赖
cd client && npm install
cd ../server && npm install

# 启动
cd server && npm run dev
# 新终端
cd ../client && npm run dev

# 访问 http://localhost:5173
